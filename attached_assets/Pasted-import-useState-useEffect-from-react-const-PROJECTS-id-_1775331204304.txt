// ─── Ticker ───
function Ticker({ lines, render }) {
const [idx, setIdx] = useState(0);
useEffect(() => {
if (lines.length <= 1) return;
const t = setInterval(() => setIdx(i => (i + 1) % lines.length), 3200);
return () => clearInterval(t);
}, [lines.length]);
return render(lines[idx]);
}
